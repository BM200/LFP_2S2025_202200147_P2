public class ProgramaCompleto {
    public static void main(String[] args) {
        // Declaración de variables de diferentes tipos
        int numero = 42;
        double decimal = 3.14;
        char letra = 'A';
        String mensaje = "Hola Mundo";
        boolean activo = true;
        
        /* Este es un comentario
           de múltiples líneas para
           probar el analizador léxico */
        
        // Operaciones aritméticas
        int suma = 10 + 5;
        int resta = 20 - 8;
        int multiplicacion = 6 * 7;
        int division = 100 / 4;
        int modulo = 15 % 4;
        
        // Impresión de valores
        System.out.println("=== PROGRAMA DE PRUEBA COMPLETO ===");
        System.out.println(mensaje);
        System.out.println("Numero: " + numero);
        System.out.println("Decimal: " + decimal);
        System.out.println("Letra: " + letra);
        
        // Estructura IF-ELSE
        if (numero > 40) {
            System.out.println("El numero es mayor a 40");
        } else {
            System.out.println("El numero es menor o igual a 40");
        }
        
        // Operadores lógicos
        boolean condicion1 = (numero > 30) && (decimal < 5.0);
        boolean condicion2 = (letra == 'A') || (letra == 'B');
        boolean condicion3 = !activo;
        
        if (condicion1) {
            System.out.println("Condicion 1 es verdadera");
        }
        
        if (condicion2) {
            System.out.println("Condicion 2 es verdadera");
        }
        
        // Bucle FOR - incremento
        System.out.println("Contando del 0 al 4:");
        for (int i = 0; i < 5; i++) {
            System.out.println(i);
        }
        
        // Bucle FOR - decremento
        System.out.println("Contando del 10 al 6:");
        for (int j = 10; j > 5; j--) {
            System.out.println(j);
        }
        
        // Bucle WHILE
        int contador = 0;
        System.out.println("Usando WHILE:");
        while (contador < 3) {
            System.out.println("Contador: " + contador);
            contador++;
        }
        
        // Bucle DO-WHILE
        int valor = 0;
        System.out.println("Usando DO-WHILE:");
        do {
            System.out.println("Valor: " + valor);
            valor++;
        } while (valor < 3);
        
        // Comparaciones múltiples
        if (suma > resta) {
            System.out.println("Suma es mayor que resta");
        }
        
        if (multiplicacion >= division) {
            System.out.println("Multiplicacion es mayor o igual");
        }
        
        if (modulo != 0) {
            System.out.println("El modulo no es cero");
        }
        
        // Declaraciones múltiples
        int a, b, c;
        a = 1;
        b = 2;
        c = 3;
        
        int x = 100, y = 200, z = 300;
        
        // Expresiones complejas
        int resultado = (a + b) * c - (x / y);
        boolean complejo = ((a < b) && (c > 0)) || ((x == 100) && (y != 0));
        
        // Operadores de incremento y decremento
        numero++;
        decimal--;
        
        // IF anidado
        if (resultado > 0) {
            if (complejo) {
                System.out.println("Resultado positivo y condicion compleja verdadera");
            } else {
                System.out.println("Resultado positivo pero condicion compleja falsa");
            }
        } else {
            System.out.println("Resultado negativo o cero");
        }
        
        // Mensaje final
        System.out.println("=== FIN DEL PROGRAMA ===");
    }
}
